DOCKER

docker : общие команды
  ps : процессы
    docker ps     : контейнеры : запущенные
    docker ps -a  : контейнеры : запущенные, остановленные и завершившиеся
    docker ps -q  : ID контейнеров : запущенных
    docker ps -qa : ID контейнеров : запущенных, остановленных и завершившихся
  logs : логи
    docker logs CTN        : логи
    docker logs -t CTN     : логи + временные метки
    docker logs -n NUM CTN : логи : tail
    docker logs -f CTN     : логи : live
  exec : команды
    docker exec CTN CMD     : команда
    docker exec -d CTN CMD  : команда : фоновый режим
    docker exec -i CTN CMD  : интерактивная команда
    docker exec -it CTN CMD : интерактивная команда : TTY
  stop + kill : завершение
    docker stop CTN : SIGTERM > timeout > SIGKILL
    docker kill CTN : SIGKILL
  rm : удаление
    docker rm CTN    : удаление
    docker rm -f CTN : удаление + остановка
    docker rm -v CTN : удаление + удаление связанных анонимных томов
  run : запуск
    docker run IMG                    : запуск
    docker run -d IMG                 : запуск : фоновый режим
    docker run -i IMG CMD             : запуск + интерактивная команда
    docker run -it IMG CMD            : запуск + интерактивная команда : TTY
    docker run --name NAM IMG         : запуск + именование
    docker run -p HST_PRT:CTN_PRT IMG : запуск + проброс порта
    docker run -e KEY=VAL IMG         : запуск + экспорт переменной окружения
    docker run -v HST_PTH:CTN_PTH IMG : запуск + монтирование тома/директории
  cp : копирование
    docker cp CTN:PTH HST_PTH : контейнер > хост
    docker cp HST_PTH CTN:PTH : хост > контейнер

---
stats : анализ контейнеров
  docker stats    : статистика : запущенные
  docker stats -a : статистика : запущенные, остановленные и завершившиеся

inspect + top + port : анализ контейнера
  docker inspect CTN : манифест контейнера
  docker top CTN     : процессы в контейнере
  docker port CTN    : проброшенные порты контейнера

--- образы

images : список образов
  docker images    : образы
  docker images -a : образы + промежуточные + dangling

build + tag : сборка и тегирование образа
  docker build -t NAM:TAG PTH : сборка контейнера : dockerfile
  docker tag IMG NAM:TAG      : создание тега

pull + push : загрузка и отправка образа
  docker pull IMG:TAG     : загрузка образа
  docker push USR/IMG:TAG : отправка образа

rmi : удаление образа
  docker rmi IMG    : удаление : не используемые
  docker rmi -f IMG : удаление : даже используемые

--- сети

network ls : список сетей
  docker network ls    : сети
  docker network ls -q : идентификаторы сетей

network create + network rm : создание и удаление сети
  docker network create NET : создание сети
  docker network rm NET     : удаление сети

network connect + network disconnect : подключение и отключение сети
  docker network connect NET CTN    : подключение контейнера
  docker network disconnect NET CTN : отключение контейнера

network inspect NET : анализ сети
  docker network inspect NET : манифест сети

--- администрирование

system df : просмотр места
  docker system df    : свободное место
  docker system df -v : свободное место : подробно

system prune : системная очистка
  docker image prune     : образы без тега
  docker volume prune    : не используемые тома
  docker container prune : остановленные контейнеры
  docker system prune    : остановленные контейнеры, пустые сети, образы без тега, кеш
  docker system prune -a : остановленные контейнеры, пустые сети, образы, кеш
