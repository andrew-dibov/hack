DOCKER

docker container :
  выполняющийся образ :
  изолированная среда :
    процессы
    сетевой стек
    файловое пространство

  ls :
    docker container ls : контейнеры
    docker container ls -a : все

  stats :
    docker container stats : ресурсы
    docker container stats -a : все

  inspect :
    docker container inspect CTN

  top + port :
    docker container top CTN : процессы
    docker container port CTN : порты

  logs :
    docker container logs CTN : логи
    docker container logs -n NUM CTN : tail
    docker container logs -f CTN : live
    docker container logs -t CTN : timestamp

  run :
    docker container run IMG
    docker container run -d IMG : фон
    docker container run -i IMG CMD : интерактив
    docker container run -it IMG CMD : интерактив + TTY
    docker container run --name NAM IMG : именование контейнера
    docker container run -e KEY=VAL IMG : экспорт переменной окружения
    docker container run -p HST_PRT:CTN_PRT IMG : проброс порта
    docker container run -v HST_PTH:CTN_PTH IMG : монтирование тома/директории

  pause + unpause :
    docker container pause CTN
    docker container unpause CTN

  stop + kill :
    docker container stop CTN : SIGTERM > timeout > SIGKILL
    docker container kill CTN : SIGKILL

  rm :
    docker container rm CTN
    docker container rm -f CTN : принудительно
    docker container rm -v CTN : с анонимными томами

  exec :
    docker container exec CTN CMD
    docker container exec -d CTN CMD : фон
    docker container exec -i CTN CMD : интерактив
    docker container exec -it CTN CMD : интерактив + TTY

  cp :
    docker container cp CTN:PTH HST_PTH : контейнер > хост
    docker container cp HST_PTH CTN:PTH : хост > контейнер

  rename :
    docker container rename OLD NAM

  commit :
    docker container commit CTN NEW_IMG:TAG : образ из контейнера


docker image : шаблон контейнера из слоев < приложение + зависимости + настройки

  ls :
    docker image ls                              : образы
    docker image ls -a                           : промежуточные + dangling

  history :
    docker image history IMG                     : слои

  build + tag :
    docker image build -t NAM:TAG PTH            : сборка
    docker image tag IMG NAM:TAG                 : тегирование

  pull + push :
    docker image pull IMG:TAG                    : хост < реестр
    docker image push USR/IMG:TAG                : хост > реестр

  rm :
    docker image rm IMG                          :
    docker image rm -f IMG                       : принудительно


docker network : виртуальный уровень для связи контейнеров с внешним миром

  ls :
    docker network ls                            : сети
    docker network ls -q                         : ID

  inspect :
    docker network inspect NET                   :

  create + rm :
    docker network create NET                    :
    docker network rm NET                        :

  connect + disconnect :
    docker network connect NET CTN               : подключение контейнера
    docker network disconnect NET CTN            : отключение контейнера


docker volume : механизм хранения данных вне контейнера

  ls :
    docker volume ls                             : томы
    docker volume ls -q                          : ID

  inspect :
    docker volume inspect VOL                    :

  create + rm :
    docker volume create VOL                     :
    docker volume rm VOL                         :
    docker volume rm -f VOL                      : принудительно


docker system :

  df :
    docker system df                             :
    docker system df -v                          : подробно

  prune :
    docker system prune                          : неиспользуемые контейнеры + сети + образы без тега + кеш сборки
    docker system prune -a                       : неиспользуемые образы
    docker system prune --volumes                : анонимные тома


DOCKER COMPOSE

docker compose : определение и запуск многоконтейнерных проектов

  ls :
    docker compose ls                            : проекты
    docker compose ls -a                         : запущенные, остановленные, завершившиеся

  ps :
    docker compose ps                            : контейнеры
    docker compose ps -a                         : запущенные, остановленные, завершившиеся

  images :
    docker compose images                        :

  top :
    docker compose top                           : процессы

  stats :
    docker compose stats                         :

  logs :
    docker compose logs                          :
    docker compose logs SVC                      :
    docker compose logs -f                       : live

  exec :
    docker compose exec SVC CMD                  :
    docker compose exec -d SVC CMD               : фон
    docker compose exec -i SVC CMD               : интерактив
    docker compose exec -it SVC CMD              : интерактив + TTY

  down :
    docker compose down                          :
    docker compose down -v                       : с анонимными томами

  start + restart + pause + unpause + stop :
    docker compose start                         :
    docker compose restart                       :
    docker compose pause                         :
    docker compose unpause                       :
    docker compose stop                          :

  up :
    docker compose up                            :
    docker compose up -d                         : фон
    docker compose up --build                    : с пересборкой
    docker compose up --force-recreate           : с пересозданием

  config :
    docker compose config                        : проверка
    docker compose config --quiet                : проверка без вывода

  build :
    docker compose build                         : сборка
    docker compose build --no-cache              : сборка без кеша

  pull :
    docker compose pull                          : обновление

  rm :
    docker compose rm                            :
    docker compose rm -f                         : принудительно


DOCKER SWARM : оркестратор : масштабирование, отказоустойчисовть и балансировка нагрузки

docker context : набор параметров подключения к кластеру

  ls :
    docker context ls                            : контексты

  inspect :
    docker context inspect                       :

  use :
    docker context use NAM                       :


docker stack : набор сервисов

  ls :
    docker stack ls                              : стеки

  ps :
    docker stack ps STK                          : задачи

  services :
    docker stack services STK                    : сервисы

  deploy :
    docker stack deploy -c FLE STK               : развертывание

  rm :
    docker stack rm STK                          :

  config :
    docker stack config -c FLE                   : проверка


docker service : определение желаемого состояния : образ, реплики, сети, тома и политика обновления

  ls :
    docker service ls                            : сервисы

  ps :
    docker service ps SVC                        : реплики

  logs :
    docker service logs SVC                      :

  inspect :
    docker service inspect SVC                   :

  update :
    docker service update --force SVC            : редеплой
    docker service update --image IMG SVC        : обновление образа

  scale :
    docker service scale SVC=NUM                 : масштабирование

  rollback :
    docker service rollback SVC                  : откат

  rm :
    docker service rm SVC                        :


docker node : хост в кластере : менеджер или рабочий

  ls :
    docker node ls                               : узлы
    docker node ls -q                            : ID

  ps :
    docker node ps NDE                           : задачи

  inspect :
    docker node inspect NDE                      :

  update :
    docker node update --availability drain NDE  : вывод из обслуживания
    docker node update --availability active NDE : ввод в обслуживание
    docker node update --label-add KEY=VAL NDE   : добавление метки
    docker node update --label-rm KEY=VAL NDE    : удаление метки
