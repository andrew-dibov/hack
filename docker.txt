DOCKER

ls  : list
df  : disk free
ps  : process status
top : table of processes

docker container :

  ls :
    docker container ls     :
    docker container ls -q  : ID
    docker container ls -a  : запущенные, остановленные, завершившиеся
    docker container ls -qa : ID : запущенные, остановленные, завершившиеся

  stats :
    docker container stats    :
    docker container stats -a : запущенные, остановленные, завершившиеся

  inspect :
    docker container inspect CTN : подробности

  top + port :
    docker container top CTN  :
    docker container port CTN : пробросы

  logs :
    docker container logs CTN        :
    docker container logs -n NUM CTN : tail
    docker container logs -f CTN     : live
    docker container logs -t CTN     : timestamp

  run :
    docker container run IMG                    :
    docker container run -d IMG                 : фоном
    docker container run -i IMG CMD             : интерактивно
    docker container run -it IMG CMD            : интерактивно : TTY
    docker container run --name NAM IMG         : именование
    docker container run -e KEY=VAL IMG         : экспорт переменной
    docker container run -p HST_PRT:CTN_PRT IMG : проброс порта
    docker container run -v HST_PTH:CTN_PTH IMG : монтирование тома/директории

  pause + unpause :
    docker container pause CTN   :
    docker container unpause CTN :

  stop + kill:
    docker container stop CTN : SIGTERM > timeout > SIGKILL
    docker container kill CTN : SIGKILL

  rm :
    docker container rm CTN    :
    docker container rm -f CTN : принудительно
    docker container rm -v CTN : анонимные тома

  exec :
    docker container exec CTN CMD     :
    docker container exec -d CTN CMD  : фоном
    docker container exec -i CTN CMD  : интерактивно
    docker container exec -it CTN CMD : интерактивно : TTY

  cp :
    docker container cp CTN:PTH HST_PTH : контейнер > хост
    docker container cp HST_PTH CTN:PTH : хост > контейнер

  rename :
    docker container rename OLD NAM :

  commit :
    docker container commit CTN NEW_IMG:TAG : создать образ из контейнера


docker image :

  ls :
    docker image ls    :
    docker image ls -a : промежуточные, dangling

  history :
    docker image history IMG : слои образа

  build + tag:
    docker image build -t NAM:TAG PTH : сборка : Dockerfile
    docker image tag IMG NAM:TAG      : тегирование

  pull + push :
    docker image pull IMG:TAG     : хост < реестр
    docker image push USR/IMG:TAG : хост > реестр

  rm :
    docker image rm IMG    :
    docker image rm -f IMG : принудительно


docker network :

  ls :
    docker network ls    :
    docker network ls -q : ID

  inspect :
    docker network inspect NET : подробности

  create + rm :
    docker network create NET :
    docker network rm NET :

  connect + disconnect:
    docker network connect NET CTN : подключение
    docker network disconnect NET CTN : отключение


docker volume :

  ls :
    docker volume ls    :
    docker volume ls -q : ID

  inspect :
    docker volume inspect VOL : подробности

  create + rm:
    docker volume create VOL :
    docker volume rm VOL     :
    docker volume rm -f VOL  : принудительно


docker system :

  df :
    docker system df    :
    docker system df -v : подробно

  prune :
    docker system prune    : очистка : остановленные контейнеры, пустые сети, образы без тега, кеш сборки
    docker system prune -a : очистка : неиспользуемые образы
