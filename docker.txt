DOCKER

docker container :

  ls :
    docker container ls     : запущенные
    docker container ls -q  : ID : запущенные
    docker container ls -a  : все (запущенные, остановленные, завершившиеся)
    docker container ls -qa : ID : все

  stats :
    docker container stats    : запущенные
    docker container stats -a : все (данные только у работающих)

  inspect :
    docker container inspect CTN : подробная информация

  top :
    docker container top CTN : процессы

  port :
    docker container port CTN : проброшенные порты

  logs :
    docker container logs CTN        : логи
    docker container logs -n NUM CTN : последние N строк
    docker container logs -f CTN     : следить в реальном времени
    docker container logs -t CTN     : с метками времени

  run :
    docker container run IMG                    : запуск
    docker container run -d IMG                 : запуск в фоне
    docker container run -i IMG CMD             : интерактивный запуск
    docker container run -it IMG CMD            : интерактивный запуск с TTY
    docker container run --name NAM IMG         : задать имя
    docker container run -p HST_PRT:CTN_PRT IMG : проброс порта
    docker container run -e KEY=VAL IMG         : переменная окружения
    docker container run -v HST_PTH:CTN_PTH IMG : монтирование тома/директории

  pause :
    docker container pause CTN : приостановить

  unpause :
    docker container unpause CTN : возобновить

  stop :
    docker container stop CTN : SIGTERM → таймаут → SIGKILL

  kill :
    docker container kill CTN : SIGKILL

  rm :
    docker container rm CTN    : удалить
    docker container rm -f CTN : принудительно удалить (остановить)
    docker container rm -v CTN : удалить и удалить анонимные тома

  exec :
    docker container exec CTN CMD     : выполнить команду
    docker container exec -d CTN CMD  : выполнить в фоне
    docker container exec -i CTN CMD  : интерактивное выполнение
    docker container exec -it CTN CMD : интерактивное выполнение с TTY

  cp :
    docker container cp CTN:PTH HST_PTH : скопировать из контейнера
    docker container cp HST_PTH CTN:PTH : скопировать в контейнер

  rename :
    docker container rename OLD NAM : переименовать

  commit :
    docker container commit CTN NEW_IMG:TAG : создать образ из контейнера


docker image :

  ls :
    docker image ls    : образы
    docker image ls -a : все образы (включая промежуточные и dangling)

  history :
    docker image history IMG : слои образа

  build :
    docker image build -t NAM:TAG PTH : собрать из Dockerfile

  tag :
    docker image tag IMG NAM:TAG : присвоить тег

  rm :
    docker image rm IMG    : удалить
    docker image rm -f IMG : принудительно удалить (даже если используется)

  pull :
    docker image pull IMG:TAG : загрузить из реестра

  push :
    docker image push USR/IMG:TAG : отправить в реестр


docker network :

  ls :
    docker network ls    : сети
    docker network ls -q : ID сетей

  inspect :
    docker network inspect NET : подробная информация

  create :
    docker network create NET : создать

  rm :
    docker network rm NET : удалить

  connect :
    docker network connect NET CTN : подключить контейнер

  disconnect :
    docker network disconnect NET CTN : отключить контейнер


docker volume :

  ls :
    docker volume ls    : тома
    docker volume ls -q : ID томов

  inspect :
    docker volume inspect VOL : подробная информация

  create :
    docker volume create VOL : создать

  rm :
    docker volume rm VOL    : удалить
    docker volume rm -f VOL : принудительно удалить (даже если используется)


docker system :

  df :
    docker system df    : использование дискового пространства
    docker system df -v : подробно (по слоям и томам)

  prune :
    docker system prune    : очистить (остановленные контейнеры, пустые сети, образы без тега, кеш сборки)
    docker system prune -a : очистить всё (включая все неиспользуемые образы)
