DOCKER

docker container :

  ls :
    docker container ls                         :
    docker container ls -q                      : ID
    docker container ls -a                      : запущенные, остановленные, завершившиеся

  stats :
    docker container stats                      :
    docker container stats -a                   : запущенные, остановленные, завершившиеся

  inspect :
    docker container inspect CTN                :

  top + port :
    docker container top CTN                    :
    docker container port CTN                   : пробросы портов

  logs :
    docker container logs CTN                   :
    docker container logs -n NUM CTN            : tail
    docker container logs -f CTN                : live
    docker container logs -t CTN                : timestamp

  run :
    docker container run IMG                    :
    docker container run -d IMG                 : фон
    docker container run -i IMG CMD             : интерактив
    docker container run -it IMG CMD            : интерактив + TTY
    docker container run --name NAM IMG         : именование контейнера
    docker container run -e KEY=VAL IMG         : экспорт переменной окружения
    docker container run -p HST_PRT:CTN_PRT IMG : проброс порта
    docker container run -v HST_PTH:CTN_PTH IMG : монтирование тома/директории

  pause + unpause :
    docker container pause CTN                  :
    docker container unpause CTN                :

  stop + kill :
    docker container stop CTN                   : SIGTERM > timeout > SIGKILL
    docker container kill CTN                   : SIGKILL

  rm :
    docker container rm CTN                     :
    docker container rm -f CTN                  : принудительно
    docker container rm -v CTN                  : с анонимными томами

  exec :
    docker container exec CTN CMD               :
    docker container exec -d CTN CMD            : фон
    docker container exec -i CTN CMD            : интерактив
    docker container exec -it CTN CMD           : интерактив + TTY

  cp :
    docker container cp CTN:PTH HST_PTH         : контейнер > хост
    docker container cp HST_PTH CTN:PTH         : хост > контейнер

  rename :
    docker container rename OLD NAM             :

  commit :
    docker container commit CTN NEW_IMG:TAG     : создание образа из контейнера


docker image :

  ls :
    docker image ls                             :
    docker image ls -a                          : промежуточные + dangling

  history :
    docker image history IMG                    : слои

  build + tag :
    docker image build -t NAM:TAG PTH           : сборка
    docker image tag IMG NAM:TAG                : тегирование

  pull + push :
    docker image pull IMG:TAG                   : хост < реестр
    docker image push USR/IMG:TAG               : хост > реестр

  rm :
    docker image rm IMG                         :
    docker image rm -f IMG                      : принудительно
